tizen.alarm.remove( )

API is used to remove an alarm from the storage.

  • Remove Single Alarm

    Removes the alarm from the storage based on the given unique Id.

    Demo
    Run Code 
    					// remove all the previous alarms from storage
					tizen.alarm.removeAll();
					
					//create and add alarm to storage.
					var alarm = new tizen.AlarmRelative(2 * tizen.alarm.PERIOD_MINUTE); 
					tizen.alarm.add(alarm, "org.tizen.browser"); 
					var alarmsarray1 = tizen.alarm.getAll();
					console.log(alarmsarray1.length + " alarms present in the storage.");
					
					//Remove the alarm from storage.
					console.log("Remove alarm with id " + alarm.id);
					tizen.alarm.remove(alarm.id);
					var alarmsarray2 = tizen.alarm.getAll();
					console.log(alarmsarray2.length + " alarms present in the storage.");

tizen.alarm.removeAll( )

API is used to remove all the alarms from the storage.

  • Remove All Alarms

    Removes all the alarms from the storage.

    Demo
    Run Code 
    
					var alarmsarray1 = tizen.alarm.getAll();
					console.log(alarmsarray1.length + " alarms present in the storage.");
					
					//create and add alarm to storage.
					console.log("Add three alarms to the storage.");
					var alarm1 = new tizen.AlarmRelative(2 * tizen.alarm.PERIOD_MINUTE); 
					tizen.alarm.add(alarm1, "org.tizen.clock");
					var alarm2 = new tizen.AlarmRelative(2 * tizen.alarm.PERIOD_MINUTE); 
					tizen.alarm.add(alarm2, "org.tizen.clock");
					var alarm3 = new tizen.AlarmRelative(2 * tizen.alarm.PERIOD_MINUTE); 
					tizen.alarm.add(alarm3, "org.tizen.clock"); 
					var alarmsarray2 = tizen.alarm.getAll();
					console.log(alarmsarray2.length + " alarms present in the storage.");
					
					//Remove the alarm from storage.
					console.log("Remove all the alarms from storage.");
					tizen.alarm.removeAll();
					var alarmsarray3 = tizen.alarm.getAll();
					console.log(alarmsarray3.length + " alarms present in the storage.");
