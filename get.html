tizen.alarm.get( )

API retrieves an alarm from the storage.

  • Get Specific Alarm

    Retrieves and returns an alarm object from the storage based on the given unique id.

    Demo
    Run Code 
    					//Remove all the previous alarms.
					tizen.alarm.removeAll();
					
					//create and add larm to the storage.
					var date = new Date(2013, 0, 20, 15, 25);
					var alarm = new tizen.AlarmAbsolute(date);
					tizen.alarm.add(alarm, "org.tizen.clock"); 
					console.log("Alarm added with id: " + alarm.id );
					
					//Get the alarm with given id from storage.
					var getalarm = tizen.alarm.get(alarm.id);
					var scheduleddate = getalarm.getNextScheduledDate();
					console.log("This alarm expires on " + scheduleddate.getFullYear() + "." + 
					scheduleddate.getMonth() + 1 + "." + scheduleddate.getDate() + "  " +
					scheduleddate.getHours() + ":" + scheduleddate.getMinutes() + ":" + scheduleddate.getSeconds());

tizen.alarm.getAll( )

API retrieves all the alarms from the storage.

  • Get All Alarms

    Returns an array of alarm objects from the storage.

    Demo
    Run Code 
    				//Remove all the previous alarms.
				tizen.alarm.removeAll();

				//Create and add two alarms to storage.				
				var alarm1 = new tizen.AlarmRelative(2 * tizen.alarm.PERIOD_DAY); 
				tizen.alarm.add(alarm1, "org.tizen.clock");
				var alarm2 = new tizen.AlarmRelative(5 * tizen.alarm.PERIOD_DAY); 
				tizen.alarm.add(alarm2, "org.tizen.browser");
				
				var alarms = tizen.alarm.getAll();
				console.log(alarms.length + " alarms present in the storage.");
				
				for (var i = 0; i < alarms.length; i++) {
					var remsec = alarms[i].getRemainingSeconds();
					console.log(remsec + " sec(s) remaining before alarm with \"" + alarms[i].id + "\" id get trigger");
				}
